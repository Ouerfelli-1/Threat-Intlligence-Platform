# Simple curl commands for creating test targets
# Replace {SCOPE_ID} with actual scope ID after creating scope

# 1. CREATE SCOPE
curl -X POST "http://localhost:8000/api/v1/scopes" -H "Content-Type: application/json" -d "{\"name\":\"test-targets\",\"enabled\":true,\"description\":\"Test targets for active reconnaissance\",\"config\":{\"passive_enabled\":true,\"active_enabled\":true,\"active_features\":{\"subdomain_enumeration\":true,\"port_scanning\":true,\"http_probing\":true,\"cve_lookup\":true},\"nmap\":{\"enabled\":true,\"scan_type\":\"fast\",\"threads\":5,\"timeout\":600,\"check_cves\":true}}}"

# 2. ADD TARGET: scanme.nmap.org
curl -X POST "http://localhost:8000/api/v1/scopes/{SCOPE_ID}/targets" -H "Content-Type: application/json" -d "{\"type\":\"domain\",\"value\":\"scanme.nmap.org\",\"enabled\":true,\"description\":\"Nmap official test target\"}"

# 3. ADD TARGET: example.com
curl -X POST "http://localhost:8000/api/v1/scopes/{SCOPE_ID}/targets" -H "Content-Type: application/json" -d "{\"type\":\"domain\",\"value\":\"example.com\",\"enabled\":true,\"description\":\"IANA example domain\"}"

# 4. ADD TARGET: owasp.org
curl -X POST "http://localhost:8000/api/v1/scopes/{SCOPE_ID}/targets" -H "Content-Type: application/json" -d "{\"type\":\"domain\",\"value\":\"owasp.org\",\"enabled\":true,\"description\":\"OWASP Foundation\"}"

# 5. CREATE SCHEDULE (every 6 hours)
curl -X POST "http://localhost:8000/api/v1/scopes/{SCOPE_ID}/schedules" -H "Content-Type: application/json" -d "{\"name\":\"test-targets-active-scan\",\"mode\":\"active\",\"cron_expression\":\"0 */6 * * *\",\"enabled\":true,\"description\":\"Active scan every 6 hours\"}"

# 6. TRIGGER IMMEDIATE ACTIVE JOB
curl -X POST "http://localhost:8000/api/v1/jobs?scope_id={SCOPE_ID}&mode=active"

# 7. LIST ALL JOBS
curl "http://localhost:8000/api/v1/jobs"

# 8. LIST ALL SCOPES
curl "http://localhost:8000/api/v1/scopes"

# 9. LIST TARGETS FOR SCOPE
curl "http://localhost:8000/api/v1/scopes/{SCOPE_ID}/targets"

# 10. LIST SCHEDULES FOR SCOPE
curl "http://localhost:8000/api/v1/scopes/{SCOPE_ID}/schedules"
